    root(values: [{rootInt1 : 111}, {rootInt2 : -999}], filter: eq(rootInt1, "value")) {

    }

    root(values: [{rootInt1 : 111}, {rootInt2 : -999}], filter: eq(rootInt1, "value")) {

    }

    root(values: [{rootInt1 : 111}, {rootInt2 : -999}], filter: eq(rootInt1, "value")) {
        child(values: [{childBool : true}])
    }
}
